const SPREADSHEET_ID = "1Fnvbd2_oDlZ_JZ874smNhDoXDqvZhOzApjAFleeZIdU"; 

function doGet(e) {
  const action = e.parameter.action;
  
  if (action == "login") {
    return checkLogin(e.parameter.id);
  } else if (action == "getSettings") {
    return getSettings();
  } else if (action == "log") {
    return recordLog(e.parameter);
  } else if (action == "getStudentRecords") { 
    // [추가] 학생별 상세 기록 조회 기능
    return getStudentRecords(e.parameter.num);
  }
  
  // 기본: 전체 학생 명단 반환 (기록 건수 포함)
  return getAllStudents();
}

function doPost(e) {
  const p = e.parameter;
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  
  // 1. 다중(일괄) 기록 저장
  if (p.action === "bulkRecord") {
    const sheet = ss.getSheetByName("기록"); // 주의: "생활기록"이 아니라 "기록"인지 확인 필요. 아래에는 "생활기록"으로 씀. 일관성 필요.
    // 기존 코드 유지하되 시트명 확인 필요해보임. 일단 사용자 코드를 존중.
    const targets = JSON.parse(p.targets); 
    
    const now = new Date();
    const timeStr = Utilities.formatDate(now, "Asia/Seoul", "yyyy-MM-dd HH:mm");

    targets.forEach(stu => {
      // 여기서는 "기록" 시트에 저장? 아래 단일 저장은 "생활기록" 시트임. 
      // 만약 같은 시트를 써야 한다면 수정 필요. 일단 사용자 원본 유지.
      sheet.appendRow([
        timeStr, 
        stu.num, 
        stu.name, 
        p.teacher, 
        p.good, 
        p.bad, 
        p.detail
      ]);
    });
    return ContentService.createTextOutput(JSON.stringify({result:"success", count: targets.length})).setMimeType(ContentService.MimeType.JSON);
  } 
  // 2. 로그 기록
  else if (p.action === "log") {
    return recordLog(p);
  }
  // 3. 기록 삭제
  else if (p.action === "delete") {
     const sheet = ss.getSheetByName("생활기록");
     const data = sheet.getDataRange().getValues();
     for (let i = 1; i < data.length; i++) {
        // B열(인덱스 1)이 학번, A열(인덱스 0)이 시간
        if (String(data[i][1]) === String(p.num) && String(data[i][0]).includes(p.time.substring(0,10))) {
           sheet.deleteRow(i + 1);
           return ContentService.createTextOutput(JSON.stringify({result:"success"})).setMimeType(ContentService.MimeType.JSON);
        }
     }
     return ContentService.createTextOutput(JSON.stringify({result:"fail"})).setMimeType(ContentService.MimeType.JSON);
  }
  // 4. 단일 기록 저장
  else {
    const sheet = ss.getSheetByName("생활기록");
    // 순서: [시간, 학번, 이름, 교사, 잘한일, 못한일, 상세]
    sheet.appendRow([p.time, p.num, p.name, p.teacher, p.good, p.bad, p.detail]);
    return ContentService.createTextOutput(JSON.stringify({result:"success"})).setMimeType(ContentService.MimeType.JSON);
  }
}

// 교사 로그인 확인
function checkLogin(id) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("Teachers"); // 시트명 대소문자 주의
  const data = sheet.getDataRange().getValues();
  let result = { success: false, name: "" };
  
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(id)) {
      result.success = true;
      result.name = data[i][1];
      break;
    }
  }
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

// 설정(칭찬/벌점) 가져오기
function getSettings() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("Settings");
  const data = sheet.getDataRange().getValues();
  let settings = { good: [], bad: [] };
  
  for (let i = 1; i < data.length; i++) {
    if(data[i][0]) settings.good.push(data[i][0]);
    if(data[i][1]) settings.bad.push(data[i][1]);
  }
  return ContentService.createTextOutput(JSON.stringify(settings)).setMimeType(ContentService.MimeType.JSON);
}

// 로그 기록
function recordLog(p) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName("AccessLog");
  if (!sheet) sheet = ss.insertSheet("AccessLog");
  
  sheet.appendRow([new Date(), p.id, p.action, p.device]); 
  return ContentService.createTextOutput(JSON.stringify({result:"logged"})).setMimeType(ContentService.MimeType.JSON);
}

// [수정] 전체 학생 데이터 (기록 건수 카운트 기능 추가)
function getAllStudents() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("학생목록");
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const students = [];

  // [추가된 로직] 생활기록 데이터 미리 스캔해서 학번별로 세기
  let recordCounts = {};
  const recordSheet = ss.getSheetByName("생활기록");
  if (recordSheet) {
    const rData = recordSheet.getDataRange().getValues();
    // doPost에서 저장 순서가 [time, num, name...] 이므로 인덱스 1이 학번입니다.
    for (let i = 0; i < rData.length; i++) {
       // 데이터가 있는 행만 체크
       if (rData[i][1]) {
           let rNum = String(rData[i][1]);
           recordCounts[rNum] = (recordCounts[rNum] || 0) + 1;
       }
    }
  }

  for (let i = 1; i < data.length; i++) {
    let student = {};
    for (let j = 0; j < headers.length; j++) {
      student[headers[j]] = data[i][j];
    }
    
    // [추가] 학생별 기록 건수 삽입
    // student 객체에 '학번'이나 '번호'가 있다고 가정
    let sNum = String(student['학번'] || student['번호'] || "");
    student['recordCount'] = recordCounts[sNum] || 0;
    
    students.push(student);
  }
  return ContentService.createTextOutput(JSON.stringify(students)).setMimeType(ContentService.MimeType.JSON);
}

// [추가] 특정 학생 기록 조회 (앱에서 상세화면 들어갈 때 필요)
function getStudentRecords(num) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("생활기록");
  
  if (!sheet) return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);

  const data = sheet.getDataRange().getValues();
  const records = [];
  
  // 데이터 구조: [시간, 학번, 이름, 교사, 잘한일, 못한일, 상세]
  // 인덱스: 0, 1, 2, 3, 4, 5, 6
  for (let i = 0; i < data.length; i++) {
    if (String(data[i][1]) === String(num)) {
       records.push({
         time: data[i][0],
         num: data[i][1],
         name: data[i][2],
         teacher: data[i][3],
         good: data[i][4],
         bad: data[i][5],
         detail: data[i][6]
       });
    }
  }
  return ContentService.createTextOutput(JSON.stringify(records)).setMimeType(ContentService.MimeType.JSON);
}
