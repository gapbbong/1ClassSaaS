const SPREADSHEET_ID = "1Fnvbd2_oDlZ_JZ874smNhDoXDqvZhOzApjAFleeZIdU"; 

function doGet(e) {
  const action = e.parameter.action;
  
  if (action == "login") {
    return checkLogin(e.parameter.id);
  } else if (action == "getSettings") {
    return getSettings();
  } else if (action == "log") {
    return recordLog(e.parameter);
  } else if (action == "getStudentRecords") { 
    return getStudentRecords(e.parameter.num);
  } else if (action == "getGroupRecords") {
    return getGroupRecords(e.parameter.grade, e.parameter.class);
  } else if (action == "verifyStudent") {
    return verifyStudent(e.parameter.num);
  }
  
  // 기본: 전체 학생 명단 반환 (기록 건수 포함)
  return getAllStudents();
}

function doPost(e) {
  const p = e.parameter;
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  
  // 1. 다중(일괄) 기록 저장
  if (p.action === "bulkRecord") {
    const sheet = ss.getSheetByName("생활기록");
    // [중요] 컬럼 순서: 학번, 이름, 잘한일, 못한일, 시간, 교사, 자세히
    const targets = JSON.parse(p.targets); 
    const now = new Date();
    const timeStr = Utilities.formatDate(now, "Asia/Seoul", "yyyy-MM-dd HH:mm");

    targets.forEach(stu => {
      sheet.appendRow([
        stu.num,      // A: 학번
        stu.name,     // B: 이름
        p.good,       // C: 잘한일
        p.bad,        // D: 못한일
        timeStr,      // E: 시간
        p.teacher,    // F: 교사
        p.detail      // G: 자세히
      ]);
    });
    return ContentService.createTextOutput(JSON.stringify({result:"success", count: targets.length})).setMimeType(ContentService.MimeType.JSON);
  } 
  
  // 2. 기록 삭제
  else if (p.action === "delete") {
     const sheet = ss.getSheetByName("생활기록");
     const data = sheet.getDataRange().getValues();
     for (let i = 1; i < data.length; i++) {
        // A열(인덱스 0)이 학번, E열(인덱스 4)이 시간
        if (String(data[i][0]) === String(p.num) && String(data[i][4]).includes(p.time.substring(0,10))) {
           sheet.deleteRow(i + 1);
           return ContentService.createTextOutput(JSON.stringify({result:"success"})).setMimeType(ContentService.MimeType.JSON);
        }
     }
     return ContentService.createTextOutput(JSON.stringify({result:"fail"})).setMimeType(ContentService.MimeType.JSON);
  }
  
  // 3. 단일 기록 저장 (기본)
  else {
    const sheet = ss.getSheetByName("생활기록");
    // [중요] 컬럼 순서 맞춤: A:학번, B:이름, C:잘한일, D:못한일, E:시간, F:교사, G:자세히
    sheet.appendRow([
        p.num, 
        p.name, 
        p.good, 
        p.bad, 
        p.time, // 클라이언트에서 시간 보내줌
        p.teacher, 
        p.detail
    ]);
    return ContentService.createTextOutput(JSON.stringify({result:"success"})).setMimeType(ContentService.MimeType.JSON);
  }
  
  // 4. 학생 기초조사 정보 업데이트
  else if (p.action === "updateStudentInfo") {
    return updateStudentInfo(p);
  }
}

// 교사 로그인 확인
function checkLogin(id) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("Teachers");
  const data = sheet.getDataRange().getValues();
  let result = { success: false, name: "" };
  
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(id)) {
      result.success = true;
      result.name = data[i][1];
      break;
    }
  }
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

// 로그 기록
function recordLog(p) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName("AccessLog");
  if (!sheet) sheet = ss.insertSheet("AccessLog");
  
  sheet.appendRow([new Date(), p.id, p.action, p.device]); 
  return ContentService.createTextOutput(JSON.stringify({result:"logged"})).setMimeType(ContentService.MimeType.JSON);
}

// 전체 학생 데이터 (기록 건수 포함)
function getAllStudents() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("학생목록");
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const students = [];

  // [수정] 생활기록 시트 컬럼 순서 반영 (A열: 학번)
  let recordCounts = {};
  const recordSheet = ss.getSheetByName("생활기록");
  if (recordSheet) {
    const rData = recordSheet.getDataRange().getValues();
    // 2번째 줄(인덱스 1)부터 데이터라고 가정
    for (let i = 1; i < rData.length; i++) {
       // A열(인덱스 0)이 학번
       if (rData[i][0]) {
           let rNum = String(rData[i][0]);
           recordCounts[rNum] = (recordCounts[rNum] || 0) + 1;
       }
    }
  }

  for (let i = 1; i < data.length; i++) {
    let student = {};
    for (let j = 0; j < headers.length; j++) {
      student[headers[j]] = data[i][j];
    }
    
    // 학생 객체에 recordCount 추가 (학번 기준)
    let sNum = String(student['학번'] || student['번호'] || "");
    student['recordCount'] = recordCounts[sNum] || 0;
    
    students.push(student);
  }
  return ContentService.createTextOutput(JSON.stringify(students)).setMimeType(ContentService.MimeType.JSON);
}

// [수정] 특정 학생 기록 조회 (컬럼 순서 반영)
function getStudentRecords(num) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("생활기록");
  
  if (!sheet) return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);

  const data = sheet.getDataRange().getValues();
  const records = [];
  
  // 데이터 구조 (스크린샷 기준):
  // A(0): 학번, B(1): 이름, C(2): 잘한일, D(3): 못한일, E(4): 시간, F(5): 교사, G(6): 자세히
  for (let i = 1; i < data.length; i++) {
    // 학번 비교 (A열)
    if (String(data[i][0]) === String(num)) {
       records.push({
         num: data[i][0],
         name: data[i][1],
         good: data[i][2],
         bad: data[i][3],
         time: data[i][4], // 날짜 객체일 수 있음
         teacher: data[i][5],
         detail: data[i][6]
       });
    }
  }
  
  // 최신순 정렬 (선택사항, 시간 기준 내림차순)
  records.sort((a, b) => new Date(b.time) - new Date(a.time));
  
  return ContentService.createTextOutput(JSON.stringify(records)).setMimeType(ContentService.MimeType.JSON);
}

function getSettings() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("Settings");
  
  const settings = {
    result: "ok",
    good: [],
    bad: []
  };
  
  if (sheet) {
    const data = sheet.getDataRange().getValues();
    // 첫 번째 줄(헤더) 제외하고 2번째 줄부터 읽기
    for (let i = 1; i < data.length; i++) {
      // A열(0): 잘한 일
      if (data[i][0]) settings.good.push(data[i][0]);
      // B열(1): 못한 일
      if (data[i][1]) settings.bad.push(data[i][1]);
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify(settings)).setMimeType(ContentService.MimeType.JSON);
}

function getGroupRecords(grade, classNum) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const recordSheet = ss.getSheetByName("생활기록");
  const studentSheet = ss.getSheetByName("학생목록");
  
  if (!recordSheet || !studentSheet) return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);

  const rData = recordSheet.getDataRange().getValues();
  const sData = studentSheet.getDataRange().getValues();
  const sHeaders = sData[0];
  
  const studentMap = {};
  for (let i = 1; i < sData.length; i++) {
    let s = {};
    for (let j = 0; j < sHeaders.length; j++) {
      s[sHeaders[j]] = sData[i][j];
    }
    const num = String(s['학번'] || s['번호'] || "");
    studentMap[num] = s;
  }

  const results = [];
  for (let i = 1; i < rData.length; i++) {
    const num = String(rData[i][0]);
    const studentInfo = studentMap[num] || {};
    
    const sGrade = String(studentInfo['학년'] || "");
    const sClass = String(studentInfo['반'] || "");

    const matchGrade = !grade || sGrade === String(grade);
    const matchClass = !classNum || sClass === String(classNum);

    if (matchGrade && matchClass) {
      results.push({
        num: rData[i][0],
        name: rData[i][1],
        good: rData[i][2],
        bad: rData[i][3],
        time: rData[i][4],
        teacher: rData[i][5],
        detail: rData[i][6],
        grade: sGrade,
        class: sClass,
        photo: studentInfo['사진저장링크'] || studentInfo['사진'] || ""
      });
    }
  }

  results.sort((a, b) => String(a.num).localeCompare(String(b.num)));
  return ContentService.createTextOutput(JSON.stringify(results)).setMimeType(ContentService.MimeType.JSON);
}


// [추가] 학생 본인 확인 (이름 마스킹 제공)
function verifyStudent(num) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("학생목록");
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    // 학번(D열, 인덱스 3) 비교
    if (String(data[i][3]) === String(num)) {
      const fullName = String(data[i][4]); // E열(인덱스 4)이 이름
      const maskedName = fullName.substring(0, 1) + "*" + (fullName.length > 2 ? fullName.substring(fullName.length - 1) : "");
      return ContentService.createTextOutput(JSON.stringify({ success: true, name: maskedName })).setMimeType(ContentService.MimeType.JSON);
    }
  }
  return ContentService.createTextOutput(JSON.stringify({ success: false })).setMimeType(ContentService.MimeType.JSON);
}

// [추가] 학생 정보 업데이트 (Survey 제출)
function updateStudentInfo(p) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName("학생목록");
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  
  let rowIndex = -1;
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][3]) === String(p.num)) {
      rowIndex = i + 1;
      break;
    }
  }
  
  if (rowIndex === -1) return ContentService.createTextOutput(JSON.stringify({ result: "fail", message: "학생을 찾을 수 없습니다." })).setMimeType(ContentService.MimeType.JSON);
  
  const surveyData = JSON.parse(p.surveyData);
  
  // 헤더를 순회하며 surveyData에 해당 키가 있으면 값을 채움
  for (let j = 0; j < headers.length; j++) {
    const headerName = headers[j].trim();
    if (surveyData[headerName] !== undefined) {
      sheet.getRange(rowIndex, j + 1).setValue(surveyData[headerName]);
    }
  }
  
  return ContentService.createTextOutput(JSON.stringify({ result: "success" })).setMimeType(ContentService.MimeType.JSON);
}
